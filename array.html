(ordenado[meio - 1] + ordenado[meio]) / 2 : ordenado[meio]; // Moda let freq = {}; let moda = []; let maxFreq = 0; numeros.forEach(num => { freq[num] = (freq[num] || 0) + 1; if (freq[num] > maxFreq) { maxFreq = freq[num]; } }); for (let num in freq) { if (freq[num] === maxFreq && maxFreq > 1) { moda.push(Number(num)); } } let positivos = numeros.filter(n => n > 0).length; let negativos = numeros.filter(n => n < 0).length; let zeros = numeros.filter(n => n === 0).length; let pares = numeros.filter(n => n % 2 === 0).length; let impares = numeros.length - pares; let percPares = (pares / numeros.length) * 100; let percImpares = (impares / numeros.length) * 100; let desvios = numeros.map(n => n - media); let ordenadoManual = [...numeros]; for (let i = 0; i < ordenadoManual.length; i++) { for (let j = 0; j < ordenadoManual.length - 1; j++) { if (ordenadoManual[j] > ordenadoManual[j + 1]) { let temp = ordenadoManual[j]; ordenadoManual[j] = ordenadoManual[j + 1]; ordenadoManual[j + 1] = temp; } } } let semDuplicados = [...new Set(numeros)]; let quadrado = numeros.map(n => n ** 2); let maioresQueMedia = numeros.filter(n => n > media); let menor = Math.min(...numeros); let semMenor = numeros.filter(n => n !== menor); let inserir100 = [...numeros]; inserir100.splice(1, 0, 100); let substituir = [...numeros]; if (substituir.length >= 5) { substituir[4] = 999; } let copiaSem3 = numeros.slice(3); console.log("\n=== RELATÓRIO ==="); console.log("\n-- Estatísticas --"); console.log("Soma:", soma); console.log("Média:", media); console.log("Mediana:", mediana); console.log("Moda:", moda.length ? moda : "Sem moda"); console.log("\n-- Análises --"); console.log("Positivos:", positivos); console.log("Negativos:", negativos); console.log("Zeros:", zeros); console.log("Pares (%):", percPares.toFixed(2) + "%"); console.log("Ímpares (%):", percImpares.toFixed(2) + "%"); console.log("Desvios:", desvios); console.log("\n-- Manipulação --"); console.log("Ordenado (manual):", ordenadoManual); console.log("Sem duplicados:", semDuplicados); console.log("Ao quadrado:", quadrado); console.log("Maiores que a média:", maioresQueMedia); console.log("\n-- Extra --"); console.log("Sem menor:", semMenor); console.log("Inserir 100:", inserir100); console.log("Substituir posição 5:", substituir); console.log("Remover 3 primeiros:", copiaSem3);
